355 /**
356  * @brief Updates the songs with edits made in the dialog's table.
357  *
358  * When the user edits an entry in the Song list, a @link SongListViewerWindow::song_edit struct@endlink
359  * is used to keep track of what changed. This function goes through all of the marked edits and updates
360  * the @link Song songs@endlink in the list. This function is only called right before the window is closed
361  * if the user clicked the Ok button and either edits occurred or songs were imported. We only edit the Song
362  * list in this function because if the edits were applied to the Song at the time when the user makes them,
363  * these edits would be kept even if the window was cancelled.
364  */
365 void SongListViewerWindow::updateSongListFromTable()
366 {
367  // See if we need to update any metadata.
368  if(mEditsOccurred)
369  {
370  // The keys in the map correspond to a song's placement in the song list.
371  // Get them in descending order so that we can modify the song list
372  // without worrying about indices changing.
373  QList<int> songListIndices = mSongEdits.keys();
374  std::sort(songListIndices.begin(), songListIndices.end(), std::greater<int>());
375  QMutableListIterator<int> iter(songListIndices);
376 
377  // Declare variables.
378  int index = 0;
379  QSpinBox* trackNumberSpinBox = nullptr;
380 
381 
382  // Handle all of the song edits.
383  while(iter.hasNext())
384  {
385  index = iter.next();
386 
387  // See if we need to remove the song. Other edits for the song don't matter if we're
388  // removing it anyway.
389  if(mSongEdits[index].remove_song)
390  {
391  Song* songToRemove = mSongList->takeAt(index);
392  Q_ASSERT_X(songToRemove != nullptr, "SongListViewerWindow::updateSongListFromTable", "Somehow we're trying to delete a null song!");
393  delete songToRemove;
394  }
395  else
396  {
397  // Update the metadata of the song as needed.
398  if(mSongEdits[index].artist_edited)
399  {
400  (*mSongList)[index]->setArtistName(ui->songListTableWidget->item(index, ARTIST_COLUMN)->text());
401  }
402  if(mSongEdits[index].album_edited)
403  {
404  (*mSongList)[index]->setAlbumName(ui->songListTableWidget->item(index, ALBUM_COLUMN)->text());
405  }
406  if(mSongEdits[index].track_number_edited)
407  {
408  trackNumberSpinBox = (QSpinBox*)ui->songListTableWidget->cellWidget(index, TRACK_NUMBER_COLUMN);
409  (*mSongList)[index]->setTrackNumber(trackNumberSpinBox->value());
410  }
411  if(mSongEdits[index].song_name_edited)
412  {
413  (*mSongList)[index]->setSongName(ui->songListTableWidget->item(index, SONG_NAME_COLUMN)->text());
414  }
415  }
416  }
417  }
